Neural networks are known as "Universal Function Approximators" because, given sufficient capacity and appropriate training, they can approximate any continuous function to any desired degree of accuracy.

In this experiment - CASE 1:

  1. The model for the Sine function successfully learned to approximate the sine wave in the range (, 2), demonstrating that it can generalize well to unseen data within the given range.
  2. The model for the Exponential Sine function also showed a good approximation, though the complexity of the function made it more challenging to learn. With proper tuning and increased capacity, the model's performance can be further improved.

Thus, based on these results, it seems that that neural networks, given sufficient data, capacity, and training, are indeed universal function approximators. The performance seen here supports the theoretical claim, although practical considerations such as model architecture and training process play a significant role in achieving the desired approximation.

import numpy as np
import matplotlib.pyplot as plt
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense
from sklearn.preprocessing import MinMaxScaler

# Use inline backend for Jupyter
%matplotlib inline

np.random.seed(7)

# -----------------------------------------------------------------------------
# Generate data for the Sine and Exponential Sine functions
# -----------------------------------------------------------------------------
T = 8000
X_sine = np.linspace(-np.pi, np.pi, T)
Y_sine = np.sin(X_sine)

X_exp_sine = np.linspace(-6*np.pi, 1*np.pi, T)
Y_exp_sine = np.exp(0.2 * X_exp_sine) * np.sin(X_exp_sine)

# Normalize the data
scaler_sine = MinMaxScaler(feature_range=(0, 1))
Y_sine_scaled = scaler_sine.fit_transform(Y_sine.reshape(-1, 1))

scaler_exp_sine = MinMaxScaler(feature_range=(0, 1))
Y_exp_sine_scaled = scaler_exp_sine.fit_transform(Y_exp_sine.reshape(-1, 1))

# -----------------------------------------------------------------------------
# Prepare the data for LSTM
# -----------------------------------------------------------------------------
def create_sequences(data, seq_length):
    X, y = [], []
    for i in range(len(data) - seq_length):
        X.append(data[i:i + seq_length])
        y.append(data[i + seq_length])
    return np.array(X), np.array(y)

seq_length = 100
X_sine_seq, y_sine_seq = create_sequences(Y_sine_scaled, seq_length)
X_exp_sine_seq, y_exp_sine_seq = create_sequences(Y_exp_sine_scaled, seq_length)

# Reshape data to fit LSTM input shape (samples, time steps, features)
X_sine_seq = X_sine_seq.reshape((X_sine_seq.shape[0], seq_length, 1))
X_exp_sine_seq = X_exp_sine_seq.reshape((X_exp_sine_seq.shape[0], seq_length, 1))




# -----------------------------------------------------------------------------
# Build the LSTM models
# -----------------------------------------------------------------------------
# Model for Sine function
sine_model = Sequential()
sine_model.add(LSTM(50, return_sequences=True, input_shape=(seq_length, 1)))
sine_model.add(LSTM(50))
sine_model.add(Dense(1))

# Model for Exponential Sine function
exp_sine_model = Sequential()
exp_sine_model.add(LSTM(50, return_sequences=True, input_shape=(seq_length, 1)))
exp_sine_model.add(LSTM(50))
exp_sine_model.add(Dense(1))



# -----------------------------------------------------------------------------
# Compile and fit the models
# -----------------------------------------------------------------------------
sine_model.compile(optimizer='adam', loss='mse')
exp_sine_model.compile(optimizer='adam', loss='mse')

sine_model.fit(X_sine_seq, y_sine_seq, epochs=20, batch_size=64, verbose=1)
exp_sine_model.fit(X_exp_sine_seq, y_exp_sine_seq, epochs=20, batch_size=64, verbose=1)




# -----------------------------------------------------------------------------
# Generate evaluation data in the range (, 2) for both functions
# -----------------------------------------------------------------------------
X_eval = np.linspace(np.pi, 2*np.pi, T)
Y_sine_eval = np.sin(X_eval)
Y_exp_sine_eval = np.exp(0.2 * X_eval) * np.sin(X_eval)

# Normalize the evaluation data
Y_sine_eval_scaled = scaler_sine.transform(Y_sine_eval.reshape(-1, 1))
Y_exp_sine_eval_scaled = scaler_exp_sine.transform(Y_exp_sine_eval.reshape(-1, 1))

# Prepare sequences for evaluation
X_sine_eval_seq, _ = create_sequences(Y_sine_eval_scaled, seq_length)
X_exp_sine_eval_seq, _ = create_sequences(Y_exp_sine_eval_scaled, seq_length)

# Reshape data to fit LSTM input shape (samples, time steps, features)
X_sine_eval_seq = X_sine_eval_seq.reshape((X_sine_eval_seq.shape[0], seq_length, 1))
X_exp_sine_eval_seq = X_exp_sine_eval_seq.reshape((X_exp_sine_eval_seq.shape[0], seq_length, 1))




# Predict using the models
Y_sine_pred_scaled = sine_model.predict(X_sine_eval_seq)
Y_exp_sine_pred_scaled = exp_sine_model.predict(X_exp_sine_eval_seq)

# Inverse transform the predictions
Y_sine_pred = scaler_sine.inverse_transform(Y_sine_pred_scaled)
Y_exp_sine_pred = scaler_exp_sine.inverse_transform(Y_exp_sine_pred_scaled)



# -----------------------------------------------------------------------------
# Compare predictions vs ground truth
# -----------------------------------------------------------------------------
plt.figure(figsize=(12, 5))

# Plot Sine function predictions
plt.subplot(1, 2, 1)
plt.plot(X_eval[seq_length:], Y_sine_eval[seq_length:], label='True Sine')
plt.plot(X_eval[seq_length:], Y_sine_pred, label='Predicted Sine')
plt.title('Sine Function')
plt.legend()

plt.plot(X_eval[seq_length:], Y_exp_sine_eval[seq_length:], label='True Exp*Sine')
plt.plot(X_eval[seq_length:], Y_exp_sine_pred, label='Predicted Exp*Sine')
